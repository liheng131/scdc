# 对话追问与记忆功能修复计划

## 问题分析

通过 Docker 日志和代码检查发现以下问题：

1. **追问触发完整四阶段工作流**：日志显示每次追问都调用 `CollectorAgent`（搜索信息），说明 `startAnalysis` 方法没有区分"新对话"和"追问"模式
2. **follow-up 接口未正确实现**：后端 `workflow.py` 使用 `OrchestratorAgent` 架构，没有 `run_follow_up_stream` 方法
3. **前端代码可能未正确部署**：前端构建的 JS 文件需要验证是否包含追问逻辑

## 修复方案

### 后端修改

#### 1. `backend/app/services/workflow.py` - 新增 follow-up 支持

```python
async def run_follow_up_stream(self, state: WorkflowState, message: str, conversation_history: list) -> AsyncGenerator[str, None]:
    """运行追问工作流，不触发完整四阶段流程，直接调用 LLM 回答问题"""
    try:
        state.status = "running"
        state.current_stage = "follow_up"
        state.current_stage_index = 0
        await self._persist_stage_update(state)

        yield self._sse("stage_start", {
            "stage": "follow_up",
            "label": "追问回答",
            "icon": "💬",
            "index": 0,
            "total": 1,
        })

        response = await self._run_follow_up(state, message, conversation_history)
        
        state.stages["follow_up"] = {
            "response": response,
            "report_markdown": response,
        }
        state.result = state.stages["follow_up"]
        await self._persist_stage_update(state)

        yield self._sse("stage_complete", {
            "stage": "follow_up",
            "label": "追问回答",
        })

        state.status = "completed"
        yield self._sse("completed", {
            "workflow_id": state.workflow_id,
            "topic": state.topic,
            "response": response,
            "report_markdown": response,
        })
    except Exception as e:
        state.status = "failed"
        state.error = str(e)
        await self._persist_stage_update(state)
        yield self._sse("error", {"error": str(e)[:500]})

async def _run_follow_up(self, state: WorkflowState, message: str, conversation_history: list) -> str:
    """直接调用 ReporterAgent 进行追问回答，不经过完整流水线"""
    from app.agents.reporter import ReporterAgent
    from app.schemas.agent import ReporterInput, AnalyzerOutput, ConversationMessage

    agent = ReporterAgent()
    
    # 将 dict 类型的 conversation_history 转换为 ConversationMessage 对象
    conv_messages = []
    if conversation_history:
        for msg in conversation_history:
            conv_messages.append(ConversationMessage(role=msg.get("role", ""), content=msg.get("content", "")))
    
    reporter_input = ReporterInput(
        task_id=state.workflow_id,
        topic=state.topic,
        analyzer_output=AnalyzerOutput(
            task_id=state.workflow_id,
            success=True,
            summary="",
            insights=[],
        ),
        include_charts=False,
        dimensions=[],
        source_contents=[],
        conversation_history=conv_messages,
    )
    
    result = await agent.execute(reporter_input)
    return result.markdown_report
```

#### 2. `backend/app/api/routes/workflow.py` - 新增 follow-up 端点

```python
class FollowUpRequest(BaseModel):
    message: str = Field(..., description="追问消息内容")
    conversation_history: List[dict] = Field(
        default_factory=list,
        description="之前的对话历史，每个元素包含 role 和 content",
    )

@router.post("/follow-up")
async def follow_up_workflow(
    req: FollowUpRequest,
    current_user: User = Depends(get_current_active_user_sse),
) -> StreamingResponse:
    state = workflow_service.create_workflow(
        topic=req.message,
        max_items=0,
        dimensions=[],
    )
    return StreamingResponse(
        workflow_service.run_follow_up_stream(
            state=state,
            message=req.message,
            conversation_history=req.conversation_history,
        ),
        media_type="text/event-stream",
        headers={
            "Cache-Control": "no-cache",
            "Connection": "keep-alive",
            "X-Accel-Buffering": "no",
        },
    )
```

**注意**：`/follow-up` 路由必须放在 `/{workflow_id}/stream` 之前定义，否则会被动态路由拦截。
